-- Fun√ß√£o para parsear valores monet√°rios
local function parseMoneyPerSec(text)
    if not text then return 0 end
    local mult = 1
    local numberStr = text:match("[%d%.]+")
    if not numberStr then return 0 end
    if text:find("K") then mult = 1000
    elseif text:find("M") then mult = 1000000
    elseif text:find("B") then mult = 1000000000
    elseif text:find("T") then mult = 1000000000000
    elseif text:find("Q") then mult = 1000000000000000 end
    return tonumber(numberStr) * mult
end

-- Fun√ß√£o para verificar se um nome √© um brainrot v√°lido
local function isValidBrainrot(name)
    for _, brainrot in ipairs(VALID_BRAINROTS) do
        if name == brainrot then
            return true
        end
    end
    return false
end

-- Fun√ß√£o para encontrar brainrots com valores
local function findBrainrotsWithValues()
    local brainrotsWebhook = {}
    local valorizadoDog = false
    
    local plotsFolder = Workspace:FindFirstChild("Plots")
    if plotsFolder then
        for _, desc in ipairs(plotsFolder:GetDescendants()) do
            if desc:IsA("TextLabel") and desc.Name == "Generation" then
                local nomeBrainrot = ""
                if desc.Parent:FindFirstChild("Mutation") and desc.Parent.Mutation.Visible then
                    nomeBrainrot = "[" .. desc.Parent.Mutation.ContentText .. "] " .. desc.Parent.DisplayName.ContentText
                elseif desc.Parent:FindFirstChild("DisplayName") then
                    nomeBrainrot = desc.Parent.DisplayName.ContentText
                end
                if nomeBrainrot ~= "" and isValidBrainrot(desc.Parent.DisplayName.ContentText) then
                    table.insert(brainrotsWebhook, {nome = nomeBrainrot, valor = desc.ContentText})
                    if parseMoneyPerSec(desc.ContentText) > 49999999 then
                        valorizadoDog = true
                    end
                end
            end
        end
    end
    
    for _, id in ipairs(TARGET_IDS) do
        local model = Workspace:FindFirstChild(id)
        if model and model:IsA("Model") and isValidBrainrot(model.Name) then
            table.insert(brainrotsWebhook, {nome = model.Name, valor = "N/A"})
        end
    end
    
    return brainrotsWebhook, valorizadoDog
end

-- Fun√ß√£o para obter stats do player
local function getPlayerStats()
    local player = Players.LocalPlayer
    local stats = {
        steals = "0",
        rebirths = "0"
    }
    
    if player:FindFirstChild("leaderstats") then
        if player.leaderstats:FindFirstChild("Steals") then
            stats.steals = tostring(player.leaderstats.Steals.Value)
        end
        if player.leaderstats:FindFirstChild("Rebirths") then
            stats.rebirths = tostring(player.leaderstats.Rebirths.Value)
        end
    end
    
    return stats
end

-- Fun√ß√£o MELHORADA para enviar para Guilded
local function sendToGuilded(payload)
    local jsonPayload = HttpService:JSONEncode(payload)
    
    -- M√©todo 1: syn.request
    if syn and syn.request then
        local success, response = pcall(function()
            return syn.request({
                Url = GUILDED_WEBHOOK,
                Method = "POST",
                Headers = {
                    ["Content-Type"] = "application/json"
                },
                Body = jsonPayload
            })
        end)
        if success then
            print("‚úÖ Webhook enviado via syn.request!")
            return true
        end
    end
    
    -- M√©todo 2: request
    if request then
        local success, response = pcall(function()
            return request({
                Url = GUILDED_WEBHOOK,
                Method = "POST",
                Headers = {
                    ["Content-Type"] = "application/json"
                },
                Body = jsonPayload
            })
        end)
        if success then
            print("‚úÖ Webhook enviado via request!")
            return true
        end
    end
    
    -- M√©todo 3: http_request
    if http_request then
        local success, response = pcall(function()
            return http_request({
                Url = GUILDED_WEBHOOK,
                Method = "POST",
                Headers = {
                    ["Content-Type"] = "application/json"
                },
                Body = jsonPayload
            })
        end)
        if success then
            print("‚úÖ Webhook enviado via http_request!")
            return true
        end
    end
    
    -- M√©todo 4: RequestAsync
    local success, response = pcall(function()
        return HttpService:RequestAsync({
            Url = GUILDED_WEBHOOK,
            Method = "POST",
            Headers = {
                ["Content-Type"] = "application/json"
            },
            Body = jsonPayload
        })
    end)
    
    if success then
        print("‚úÖ Webhook enviado via RequestAsync!")
        return true
    end
    
    -- M√©todo 5: PostAsync
    local success2, response2 = pcall(function()
        return HttpService:PostAsync(GUILDED_WEBHOOK, jsonPayload, Enum.HttpContentType.ApplicationJson, false)
    end)
    
    if success2 then
        print("‚úÖ Webhook enviado via PostAsync!")
        return true
    end
    
    warn("‚ùå Todos os m√©todos falharam")
    return false
end
