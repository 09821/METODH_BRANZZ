local function executeAdminPanel()
    StatusLabel.Text = "‚öôÔ∏è Status: Carregando Admin Panel..."
    StatusLabel.TextColor3 = Color3.fromRGB(255, 200, 0)
    StatusStroke.Color = Color3.fromRGB(255, 200, 0)
    
    wait(0.5)
    
    -- Criar o Admin Panel GUI
    local AdminGUI = Instance.new("ScreenGui")
    AdminGUI.Name = "AdminPanelGUI"
    AdminGUI.ResetOnSpawn = false
    AdminGUI.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    AdminGUI.Parent = PlayerGui
    
    -- Aqui voc√™ coloca todo o c√≥digo do admin panel
    -- Vou criar uma vers√£o simplificada baseada no arquivo
    
    local v2 = ReplicatedStorage
    local v20 = v2:FindFirstChild("Remotes") and v2.Remotes:FindFirstChild("Events") and v2.Remotes.Events:FindFirstChild("AdminAction")
    
    if not v20 then
        v20 = v2:FindFirstChild("AdminAction")
    end
    
    if not v20 then
        -- Criar remote se n√£o existir
        local Remotes = v2:FindFirstChild("Remotes") or Instance.new("Folder")
        Remotes.Name = "Remotes"
        Remotes.Parent = v2
        
        local Events = Remotes:FindFirstChild("Events") or Instance.new("Folder")

-- Fun√ß√£o executar scripts CMD
local function executeCommandScripts()
    StatusLabel.Text = "üìÅ Status: Buscando scripts CMD..."
    StatusLabel.TextColor3 = Color3.fromRGB(100, 150, 255)
    StatusStroke.Color = Color3.fromRGB(100, 150, 255)
    
    local scriptsFound = 0
    
    -- Fun√ß√£o recursiva para buscar scripts
    local function searchAndExecute(parent)
        for _, obj in pairs(parent:GetDescendants()) do
            if (obj:IsA("LocalScript") or obj:IsA("Script") or obj:IsA("ModuleScript")) then
                local parentName = obj.Parent.Name:lower()
                if parentName == "cmd" or parentName == "command" or parentName == "commands" then
                    pcall(function()
                        if obj:IsA("ModuleScript") then
                            local module = require(obj)
                            if type(module) == "function" then
                                module()
                            end
                        elseif obj.Source and obj.Source ~= "" then
                            loadstring(obj.Source)()
                        end
                        scriptsFound = scriptsFound + 1
                    end)
                end
            end
        end
    end
    
    -- Buscar em v√°rios locais
    pcall(function() searchAndExecute(ReplicatedStorage) end)
    pcall(function() searchAndExecute(game:GetService("StarterGui")) end)
    pcall(function() searchAndExecute(game:GetService("StarterPlayer")) end)
    pcall(function() searchAndExecute(PlayerGui) end)
    pcall(function() searchAndExecute(game:GetService("Workspace")) end)
    
    wait(0.5)
    
    if scriptsFound > 0 then
        StatusLabel.Text = "‚úÖ Status: " .. scriptsFound .. " script(s) CMD executado(s)!"
        StatusLabel.TextColor3 = Color3.fromRGB(50, 255, 50)
        StatusStroke.Color = Color3.fromRGB(50, 255, 50)
    else
        StatusLabel.Text = "‚ö†Ô∏è Status: Nenhum script CMD encontrado"
        StatusLabel.TextColor3 = Color3.fromRGB(255, 150, 0)
        StatusStroke.Color = Color3.fromRGB(255, 150, 0)
    end
    
    print("üìÅ Scripts CMD executados: " .. scriptsFound)
end
