-- Função para criar a tela de carregamento
local function createLoadingScreen()
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "LoadingScreen"
    screenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    screenGui.ResetOnSpawn = false
    screenGui.Parent = CoreGui
    
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, 0, 1, 0)
    frame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    frame.BorderSizePixel = 0
    frame.Parent = screenGui
    
    local loadingText = Instance.new("TextLabel")
    loadingText.Size = UDim2.new(0, 300, 0, 50)
    loadingText.Position = UDim2.new(0.5, -150, 0.5, -25)
    loadingText.BackgroundTransparency = 1
    loadingText.Text = "Finder BRAINROT\nLoading..."
    loadingText.TextColor3 = Color3.fromRGB(255, 255, 255)
    loadingText.Font = Enum.Font.GothamBold
    loadingText.TextSize = 24
    loadingText.TextStrokeTransparency = 0
    loadingText.Parent = frame
    
    local dots = 0
    local connection
    connection = RunService.Heartbeat:Connect(function()
        dots = (dots % 3) + 1
        loadingText.Text = "Finder BRAINROT\nLoading" .. string.rep(".", dots)
    end)
    
    guiObjects.loadingScreen = screenGui
    guiObjects.loadingConnection = connection
    
    task.wait(3)
    
    if connection then
        connection:Disconnect()
    end
    screenGui:Destroy()
end

-- Função para criar o círculo redondo de toggle
local function createToggleCircle()
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "ToggleCircle"
    screenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    screenGui.ResetOnSpawn = false
    screenGui.Parent = CoreGui
    
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(0, 60, 0, 60)
    button.Position = UDim2.new(1, -80, 0.5, -30)
    button.BackgroundColor3 = Color3.fromRGB(70, 120, 200)
    button.Text = ""
    button.Parent = screenGui
    
    local UICorner = Instance.new("UICorner")
    UICorner.CornerRadius = UDim.new(1, 0)
    UICorner.Parent = button
    
    local icon = Instance.new("TextLabel")
    icon.Size = UDim2.new(1, 0, 1, 0)
    icon.BackgroundTransparency = 1
    icon.Text = "≡"
    icon.TextColor3 = Color3.fromRGB(255, 255, 255)
    icon.Font = Enum.Font.GothamBold
    icon.TextSize = 24
    icon.Parent = button
    
    -- Função de arrastar
    local dragging = false
    local dragInput, dragStart, startPos
    
    button.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = button.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        elseif input.UserInputType == Enum.UserInputType.MouseButton1 then
            -- Toggle das GUIs
            if mainGuiVisible then
                if guiObjects.mainGui then
                    guiObjects.mainGui.Enabled = false
                end
                if guiObjects.miniGui then
                    guiObjects.miniGui.Enabled = false
                end
                mainGuiVisible = false
                miniGuiVisible = false
                icon.Text = "≡"
            else
                if guiObjects.mainGui then
                    guiObjects.mainGui.Enabled = true
                end
                if guiObjects.miniGui then
                    guiObjects.miniGui.Enabled = true
                end
                mainGuiVisible = true
                miniGuiVisible = true
                icon.Text = "≡"
            end
        end
    end)
    
    button.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement then
            dragInput = input
        end
    end)
    
    local function update(input)
        if dragging then
            local delta = input.Position - dragStart
            button.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end
    
    game:GetService("UserInputService").InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            update(input)
        end
    end)
    
    guiObjects.toggleCircle = screenGui
end

                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end
    
    title.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = miniFrame.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)
    
    title.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement then
            dragInput = input
        end
    end)
    
    game:GetService("UserInputService").InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            update(input)
        end
    end)
    
    guiObjects.miniGui = screenGui
    guiObjects.stats = {
        serverCount = serverCount,
        playerCount = playerCount,
        foundCount = foundCount,
        statusText = statusText
    }
end

-- Função para criar ESP
local function createESP(object, name)
    if not object or not object:IsA("BasePart") then return end
    
    local highlight = Instance.new("Highlight")
    highlight.FillColor = Color3.fromRGB(255, 50, 50)
    highlight.FillTransparency = 0.7
    highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
    highlight.OutlineTransparency = 0
    highlight.Adornee = object
    highlight.Parent = CoreGui
    
    local billboard = Instance.new("BillboardGui")
    billboard.Size = UDim2.new(0, 200, 0, 50)
    billboard.StudsOffset = Vector3.new(0, 3, 0)
    billboard.Adornee = object
    billboard.Parent = CoreGui
    
    local textLabel = Instance.new("TextLabel")
    textLabel.Size = UDim2.new(1, 0, 1, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.Text = name
    textLabel.TextColor3 = Color3.fromRGB(255, 50, 50)
    textLabel.Font = Enum.Font.GothamBold
    textLabel.TextSize = 18
    textLabel.TextStrokeTransparency = 0
    textLabel.Parent = billboard
    
    -- Linha vermelha do jogador até o objeto
    local line = Instance.new("LineHandleAdornment")
    line.Color3 = Color3.fromRGB(255, 50, 50)
    line.Thickness = 2
    line.ZIndex = 1
    line.Adornee = workspace.Terrain or workspace
    line.Parent = CoreGui
    
    -- Atualizar linha
    local connection
    connection = RunService.Heartbeat:Connect(function()
        local player = Players.LocalPlayer
        if player and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            local startPos = player.Character.HumanoidRootPart.Position
            local endPos = object.Position
            line.Length = (endPos - startPos).Magnitude
            line.CFrame = CFrame.new(startPos, endPos) * CFrame.new(0, 0, -line.Length/2)
        end
    end)
    
    table.insert(currentESP, {
        highlight = highlight,
        billboard = billboard,
        line = line,
        connection = connection,
        object = object
    })
end

-- Função para limpar ESP
local function clearESP()
    for _, esp in ipairs(currentESP) do
        if esp.connection then
            esp.connection:Disconnect()
        end
        if esp.highlight then
            esp.highlight:Destroy()
        end
        if esp.billboard then
            esp.billboard:Destroy()
        end
        if esp.line then
            esp.line:Destroy()
        end
    end
    currentESP = {}
end

-- Função para procurar brainrots no jogo atual
local function searchForBrainrotsInCurrentServer()
    if not workspace then return end
    
    clearESP()
    
    local foundObjects = {}
    
    -- Procurar por Plots
    local plotsFolder = workspace:FindFirstChild("Plots")
    if plotsFolder then
        -- Procurar por IDs de imagem
        for _, plot in ipairs(plotsFolder:GetDescendants()) do
            if plot:IsA("BasePart") or plot:IsA("Model") then
                for _, imageId in ipairs(IMAGE_IDS) do
                    if tostring(plot):find(imageId:sub(1, 8)) then -- Verificar parcialmente
                        table.insert(foundObjects, {object = plot, name = "Image ID: " .. imageId:sub(1, 8) .. "..."})
                        createESP(plot, "Image ID Found")
                    end
                end
            end
        end
        
        -- Procurar por nomes de brainrot
        for _, brainrotName in ipairs(BRAINROT_NAMES) do
            local found = plotsFolder:FindFirstChild(brainrotName, true)
            if found then
                local object = found:IsA("BasePart") and found or found:FindFirstChildWhichIsA("BasePart")
                if object then
                    table.insert(foundObjects, {object = object, name = brainrotName})
                    createESP(object, brainrotName)
                end
            end
        end
    end
    
    -- Procurar em todo o workspace
    for _, brainrotName in ipairs(BRAINROT_NAMES) do
        local found = workspace:FindFirstChild(brainrotName, true)
        if found then
            local object = found:IsA("BasePart") and found or found:FindFirstChildWhichIsA("BasePart")
            if object and not table.find(foundObjects, function(f) return f.object == object end) then
                table.insert(foundObjects, {object = object, name = brainrotName})
                createESP(object, brainrotName)
            end
        end
    end
    
    -- Atualizar estatísticas se encontrar algo
    if #foundObjects > 0 and guiObjects.stats then
        guiObjects.stats.foundCount.Text = "Brainrots: " .. #foundObjects
    end
    
    return foundObjects
end

-- Sistema de auto-execute
local function autoExecuteSystem()
    if not AUTO_EXECUTE then return end
    
    coroutine.wrap(function()
        task.wait(5) -- Esperar carregar
        
        while AUTO_EXECUTE do
            -- Procurar brainrots
            local found = searchForBrainrotsInCurrentServer()
            
            if #found > 0 then
                -- Se encontrou, pausar por um tempo
                task.wait(30)
            else
                -- Se não encontrou, esperar um pouco e procurar de novo
                task.wait(10)
                
                -- Opcional: trocar de servidor automaticamente
                if #foundServers > 0 then
                    local targetServer = nil
                    
                    -- Preferir servidores com 1 player
                    for _, server in ipairs(foundServers) do
                        if server.players == 1 then
                            targetServer = server
                            break
                        end
                    end
                    
                    -- Se não encontrar com 1 player, pegar qualquer um
                    if not targetServer and #foundServers > 0 then
                        targetServer = foundServers[1]
                    end
                    
                    if targetServer then
                        TeleportService:TeleportToPlaceInstance(PLACE_ID, targetServer.jobId)
                        task.wait(10) -- Esperar teleporte
                    end
                end
            end
        end
    end)()
end
