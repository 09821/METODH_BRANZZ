local selectedModel = nil

-- Fun√ß√£o para buscar modelos
local function findModels()
    local models = {}
    
    -- Nomes de pastas poss√≠veis
    local folderNames = {
        "Things",
        "Things models",
        "ThingsModels",
        "Things Models",
        "Things Modules",
        "Things Rarity"
    }
    
    -- Buscar nas pastas
    for _, folderName in ipairs(folderNames) do
        local folder = ReplicatedStorage:FindFirstChild(folderName)
        if folder then
            -- Buscar modelos diretos
            for _, child in pairs(folder:GetChildren()) do
                if child:IsA("Model") or child:IsA("Folder") then
                    -- Se for pasta, buscar dentro
                    if child:IsA("Folder") then
                        for _, subChild in pairs(child:GetChildren()) do
                            if subChild:IsA("Model") then
                                local rarity = subChild:FindFirstChild("Rarity") or subChild:GetAttribute("Rarity") or "Unknown"
                                if typeof(rarity) ~= "string" then
                                    rarity = tostring(rarity.Value or "Unknown")
                                end
                                
                                table.insert(models, {
                                    Name = subChild.Name,
                                    Model = subChild,
                                    Rarity = rarity
                                })
                            end
                        end
                    else
                        -- Modelo direto
                        local rarity = child:FindFirstChild("Rarity") or child:GetAttribute("Rarity") or "Unknown"
                        if typeof(rarity) ~= "string" then
                            rarity = tostring(rarity.Value or "Unknown")
                        end
                        
                        table.insert(models, {
                            Name = child.Name,
                            Model = child,
                            Rarity = rarity
                        })
                    end
                end
            end
        end
    end
    
    -- Ordenar alfabeticamente
    table.sort(models, function(a, b)
        return a.Name < b.Name
    end)
    
    return models
end

-- Buscar modelos
allModels = findModels()

-- ==================== SISTEMA DE SPAWN ====================
SpawnBtn.MouseButton1Click:Connect(function()
    if not selectedModel then
        SpawnBtn.Text = "‚ùå Selecione um gato!"
        wait(1.5)
        SpawnBtn.Text = "üê± Spawn"
        return
    end
    
    -- Feedback visual
    SpawnBtn.Text = "‚è≥ Spawnando..."
    SpawnBtn.BackgroundColor3 = Color3.fromRGB(255, 180, 50)
    
    -- Encontrar Waypoint
    local waypoint = Workspace:FindFirstChild("Waypoints")
    if waypoint then
        waypoint = waypoint:FindFirstChild("Waypoint 1") or waypoint:FindFirstChild("Waypoint1")
    end
    
    if not waypoint then
        waypoint = Workspace:FindFirstChild("Waypoint 1") or Workspace:FindFirstChild("Waypoint1")
    end
    
    if not waypoint then
        SpawnBtn.Text = "‚ùå Waypoint n√£o encontrado!"
        SpawnBtn.BackgroundColor3 = Color3.fromRGB(255, 70, 70)
        wait(2)
        SpawnBtn.Text = "üê± Spawn"
        SpawnBtn.BackgroundColor3 = Color3.fromRGB(50, 150, 255)
        return
    end
    
    -- Tentar ativar RemoteEvents relacionados
    pcall(function()
        for _, remote in pairs(ReplicatedStorage:GetDescendants()) do
            if remote:IsA("RemoteEvent") or remote:IsA("RemoteFunction") then
                pcall(function()
                    if remote:IsA("RemoteEvent") then
                        remote:FireServer("SpawnThing", {
                            Name = selectedModel.Name,
                            Model = selectedModel,
                            Position = waypoint.Position
                        })
                        
                        remote:FireServer("Spawn", selectedModel.Name, waypoint.Position)
                    end
                end)
            end
        end
    end)
    
    -- Clonar e posicionar modelo
    pcall(function()
        local clone = selectedModel:Clone()
        
        -- Posicionar no waypoint
        if clone.PrimaryPart then
            clone:SetPrimaryPartCFrame(waypoint.CFrame + Vector3.new(0, 5, 0))
        elseif clone:FindFirstChildWhichIsA("BasePart") then
            clone:FindFirstChildWhichIsA("BasePart").CFrame = waypoint.CFrame + Vector3.new(0, 5, 0)
        end
        
        clone.Parent = Workspace
        
        -- Ativar scripts do modelo
        for _, script in pairs(clone:GetDescendants()) do
            if script:IsA("Script") or script:IsA("LocalScript") then
                script.Disabled = false
            end
        end
        
        print("üê± Gato spawnado: " .. selectedModel.Name .. " em Waypoint 1")
    end)
    
    -- Feedback de sucesso
    wait(0.5)
    SpawnBtn.Text = "‚úÖ Spawnado!"
    SpawnBtn.BackgroundColor3 = Color3.fromRGB(50, 255, 100)
    
    wait(1.5)
    SpawnBtn.Text = "üê± Spawn"
    SpawnBtn.BackgroundColor3 = Color3.fromRGB(50, 150, 255)
end)
